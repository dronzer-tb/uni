#!/usr/bin/env bash
# ============================================================
#  uni — Universal Package Manager Wrapper v2.0.0
#  Commands: uni install | uni remove | uni purge | uni list | uni update
# ============================================================

set -u

UNI_VERSION="2.0.0"

if [[ -n "${SUDO_USER:-}" ]]; then
    REAL_HOME=$(getent passwd "$SUDO_USER" | cut -d: -f6)
else
    REAL_HOME="$HOME"
fi

UNI_REGISTRY="$REAL_HOME/.local/share/uni/installed.json"
UNI_APPIMAGE_DIR="$REAL_HOME/.local/share/uni/appimages"
UNI_DESKTOP_DIR="$REAL_HOME/.local/share/applications"
UNI_ICON_DIR="$REAL_HOME/.local/share/uni/icons"

# ── Colors ───────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
MAGENTA='\033[0;35m'
BOLD='\033[1m'
DIM='\033[2m'
RESET='\033[0m'

# ── UI Helpers ───────────────────────────────────────────────
info()    { echo -e "${CYAN}${BOLD} → ${RESET} $*"; }
success() { echo -e "${GREEN}${BOLD} ✓ ${RESET} $*"; }
warn()    { echo -e "${YELLOW}${BOLD} ! ${RESET} $*"; }
error()   { echo -e "${RED}${BOLD} ✗ ${RESET} $*" >&2; }
die()     { error "$*"; exit 1; }
step()    { echo -e "${MAGENTA}${BOLD} ⠿ ${RESET}${DIM}$*${RESET}"; }
divider() { echo -e "${DIM}────────────────────────────────────────${RESET}"; }
header()  { echo ""; echo -e "${BOLD}${BLUE}$*${RESET}"; divider; }

# ── Spinner ──────────────────────────────────────────────────
_SPIN_PID=""
spin_start() {
    local msg="$1"
    {
        local frames=('⠋' '⠙' '⠹' '⠸' '⠼' '⠴' '⠦' '⠧' '⠇' '⠏')
        local i=0
        while true; do
            printf "\r${CYAN}${BOLD} %s ${RESET}${DIM}%s${RESET}   " "${frames[$((i % ${#frames[@]}))]}" "$msg"
            sleep 0.08
            ((i++))
        done
    } &
    _SPIN_PID=$!
    disown "$_SPIN_PID" 2>/dev/null || true
}

spin_stop() {
    if [[ -n "$_SPIN_PID" ]]; then
        kill "$_SPIN_PID" 2>/dev/null || true
        wait "$_SPIN_PID" 2>/dev/null || true
        _SPIN_PID=""
        printf "\r\033[2K"
    fi
}

trap 'spin_stop' EXIT INT TERM

in_array() {
    local needle="$1"; shift
    local e
    for e in "$@"; do [[ "$e" == "$needle" ]] && return 0; done
    return 1
}

# ── Dependency Checks ────────────────────────────────────────
command -v python3 &>/dev/null || die "python3 is required but not installed."
command -v curl    &>/dev/null || die "curl is required but not installed."

if [[ -n "${SUDO_USER:-}" ]]; then
    warn "Running uni with sudo is not recommended."
    warn "User-level backends (Flatpak) may not be found under root's environment."
fi

# ── Init registry ────────────────────────────────────────────
init_registry() {
    mkdir -p "$(dirname "$UNI_REGISTRY")" "$UNI_APPIMAGE_DIR" "$UNI_DESKTOP_DIR" "$UNI_ICON_DIR"
    [[ -f "$UNI_REGISTRY" ]] || echo '{}' > "$UNI_REGISTRY"
    if [[ -n "${SUDO_USER:-}" ]]; then
        chown -R "$SUDO_USER:$SUDO_USER" "$(dirname "$UNI_REGISTRY")" 2>/dev/null || true
    fi
}

# ── Registry helpers ─────────────────────────────────────────
registry_add() {
    local name="$1" backend="$2" id="$3"
    python3 - "$name" "$backend" "$id" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, backend, id_val, reg_path = sys.argv[1], sys.argv[2], sys.argv[3], sys.argv[4]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
d[name] = {'backend': backend, 'id': id_val}
with open(reg_path, 'w') as f: json.dump(d, f, indent=2)
PYEOF
}

registry_remove() {
    local name="$1"
    python3 - "$name" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, reg_path = sys.argv[1], sys.argv[2]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
d.pop(name, None)
with open(reg_path, 'w') as f: json.dump(d, f, indent=2)
PYEOF
}

registry_get() {
    local name="$1" field="$2"
    python3 - "$name" "$field" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, field, reg_path = sys.argv[1], sys.argv[2], sys.argv[3]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
print(d.get(name, {}).get(field, ''))
PYEOF
}

registry_list() {
    python3 - "$UNI_REGISTRY" << 'PYEOF'
import json, sys
reg_path = sys.argv[1]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
if not d:
    print('  No packages installed via uni.')
else:
    for name, info in sorted(d.items()):
        print(f"  {name:<32} [{info['backend']:<10}]  {info['id']}")
PYEOF
}

# ── Backend detection ────────────────────────────────────────
has_nala()    { command -v nala    &>/dev/null; }
has_apt()     { command -v apt     &>/dev/null; }
has_flatpak() { command -v flatpak &>/dev/null; }
has_snap()    { command -v snap    &>/dev/null; }
has_dpkg()    { command -v dpkg    &>/dev/null; }

apt_cmd() { has_nala && echo "nala" || echo "apt"; }

# ── System arch ──────────────────────────────────────────────
sys_arch() {
    case "$(uname -m)" in
        x86_64)  echo "amd64" ;;
        aarch64) echo "arm64" ;;
        armv7l)  echo "armhf" ;;
        *)       uname -m ;;
    esac
}

arch_patterns() {
    case "$(uname -m)" in
        x86_64)  echo "x86_64 amd64 x64" ;;
        aarch64) echo "aarch64 arm64" ;;
        armv7l)  echo "armv7l armhf arm" ;;
        *)       uname -m ;;
    esac
}

# ── GitHub integration ───────────────────────────────────────

parse_github_repo() {
    local input="$1"
    if [[ "$input" =~ ^https?://github\.com/([^/]+/[^/]+) ]]; then
        echo "${BASH_REMATCH[1]}" | sed 's|/releases.*||; s|\.git$||'
    elif [[ "$input" =~ ^([a-zA-Z0-9_.-]+/[a-zA-Z0-9_.-]+)$ ]]; then
        echo "$input"
    else
        echo ""
    fi
}

gh_release_json() {
    local repo="$1"
    curl -sf --max-time 10 \
        -H "Accept: application/vnd.github+json" \
        -H "X-GitHub-Api-Version: 2022-11-28" \
        "https://api.github.com/repos/${repo}/releases/latest" 2>/dev/null
}

# Score an asset name — higher = better match for this system
score_asset() {
    local name="${1,,}"
    local score=0

    # Arch matching
    for pat in $(arch_patterns); do
        if [[ "$name" == *"${pat,,}"* ]]; then
            score=$((score + 50)); break
        fi
    done

    # Penalize other arches
    local other_arches="x86_64 amd64 x64 aarch64 arm64 armv7l armhf i386 i686"
    for pat in $other_arches; do
        local is_ours=0
        for mine in $(arch_patterns); do
            [[ "${pat,,}" == "${mine,,}" ]] && is_ours=1 && break
        done
        [[ $is_ours -eq 1 ]] && continue
        [[ "$name" == *"${pat,,}"* ]] && score=$((score - 200))
    done

    # Format preference
    [[ "$name" == *.deb      ]] && score=$((score + 40))
    [[ "$name" == *.appimage ]] && score=$((score + 30))
    [[ "$name" == *.tar.gz   ]] && score=$((score + 10))
    [[ "$name" == *.tar.xz   ]] && score=$((score + 10))
    [[ "$name" == *.zip      ]] && score=$((score + 5))
    [[ "$name" == *linux*    ]] && score=$((score + 20))

    # Penalize checksums, debug, non-linux
    [[ "$name" == *.sha256  ]] && score=$((score - 500))
    [[ "$name" == *.sha512  ]] && score=$((score - 500))
    [[ "$name" == *.asc     ]] && score=$((score - 500))
    [[ "$name" == *source*  ]] && score=$((score - 100))
    [[ "$name" == *debug*   ]] && score=$((score - 100))
    [[ "$name" == *.exe     ]] && score=$((score - 300))
    [[ "$name" == *.dmg     ]] && score=$((score - 300))
    [[ "$name" == *windows* ]] && score=$((score - 300))
    [[ "$name" == *darwin*  ]] && score=$((score - 300))
    [[ "$name" == *macos*   ]] && score=$((score - 300))

    echo "$score"
}

# Parse asset list from release JSON — outputs "url\tname" per line
pick_best_asset() {
    local json="$1"
    python3 - "$json" << 'PYEOF'
import json, sys
data = json.loads(sys.argv[1])
assets = data.get('assets', [])
if not assets:
    sys.exit(1)
for a in assets:
    print(f"{a['browser_download_url']}\t{a['name']}")
PYEOF
}

install_from_github() {
    local repo="$1"

    header "GitHub  →  ${repo}"
    spin_start "Fetching latest release..."
    local json
    json=$(gh_release_json "$repo")
    spin_stop

    [[ -z "$json" ]] && die "Could not fetch release info for ${repo}. Check the repo name and your connection."

    local tag
    tag=$(python3 -c "import json,sys; d=json.loads(sys.argv[1]); print(d.get('tag_name','unknown'))" "$json" 2>/dev/null || echo "unknown")
    info "Latest release: ${BOLD}${tag}${RESET}"

    local asset_lines
    asset_lines=$(pick_best_asset "$json") || die "No release assets found for ${repo}."

    # Score all assets
    local best_url="" best_name="" best_score=-9999
    local all_urls=() all_names=()
    while IFS=$'\t' read -r url name; do
        all_urls+=("$url")
        all_names+=("$name")
        local s
        s=$(score_asset "$name")
        if (( s > best_score )); then
            best_score=$s
            best_url="$url"
            best_name="$name"
        fi
    done <<< "$asset_lines"

    if (( best_score < 0 )) || [[ -z "$best_url" ]]; then
        # No confident match — show full menu
        warn "Could not auto-detect asset for $(uname -m). Please select:"
        echo ""
        local i=1
        for name in "${all_names[@]}"; do
            echo -e "  ${BOLD}${i})${RESET} ${name}"
            ((i++))
        done
        echo ""
        local choice
        read -rp "$(echo -e "${CYAN}${BOLD}Select asset [1-${#all_names[@]}]:${RESET} ")" choice
        local idx=$(( choice - 1 ))
        best_url="${all_urls[$idx]}"
        best_name="${all_names[$idx]}"
    else
        # Show auto-selection with option to override
        echo ""
        local i=1
        for name in "${all_names[@]}"; do
            if [[ "$name" == "$best_name" ]]; then
                echo -e "  ${GREEN}${BOLD}${i})${RESET} ${GREEN}${name}${RESET}  ${DIM}← auto-selected${RESET}"
            else
                echo -e "  ${DIM}${i}) ${name}${RESET}"
            fi
            ((i++))
        done
        echo ""
        local choice
        read -rp "$(echo -e "${CYAN}${BOLD}Press Enter to confirm, or pick another [1-${#all_names[@]}]:${RESET} ")" choice
        if [[ -n "$choice" && "$choice" =~ ^[0-9]+$ ]]; then
            local idx=$(( choice - 1 ))
            if (( idx >= 0 && idx < ${#all_names[@]} )); then
                best_url="${all_urls[$idx]}"
                best_name="${all_names[$idx]}"
            fi
        fi
    fi

    info "Downloading ${BOLD}${best_name}${RESET}..."
    local lower="${best_name,,}"

    if [[ "$lower" == *.deb ]]; then
        local tmp_deb
        tmp_deb=$(mktemp --suffix=.deb)
        curl -Lf# "$best_url" -o "$tmp_deb" || { rm -f "$tmp_deb"; die "Download failed."; }
        install_deb "$tmp_deb"
        rm -f "$tmp_deb"
    elif [[ "$lower" == *.appimage ]]; then
        install_appimage "$best_url"
    elif [[ "$lower" == *.tar.gz || "$lower" == *.tar.xz || "$lower" == *.zip ]]; then
        install_archive "$best_url" "$best_name" "$repo"
    else
        die "Unsupported asset type: ${best_name}\nDownload manually: ${best_url}"
    fi
}

# Install binary from archive to ~/.local/bin
install_archive() {
    local url="$1" filename="$2" repo="$3"
    local pkg_name
    pkg_name=$(basename "$repo")

    local tmpdir
    tmpdir=$(mktemp -d)
    curl -Lf# "$url" -o "$tmpdir/$filename" || { rm -rf "$tmpdir"; die "Download failed."; }

    step "Extracting..."
    if [[ "$filename" == *.zip ]]; then
        command -v unzip &>/dev/null || { rm -rf "$tmpdir"; die "unzip required: sudo apt install unzip"; }
        unzip -q "$tmpdir/$filename" -d "$tmpdir/extracted"
    else
        mkdir -p "$tmpdir/extracted"
        tar -xf "$tmpdir/$filename" -C "$tmpdir/extracted"
    fi

    # Find executables
    local bins=()
    while IFS= read -r -d '' f; do
        bins+=("$f")
    done < <(find "$tmpdir/extracted" -maxdepth 3 -type f -executable \
        -not -name "*.so" -not -name "*.sh" -print0 2>/dev/null)

    [[ ${#bins[@]} -eq 0 ]] && { rm -rf "$tmpdir"; die "No executable found in archive."; }

    local chosen_bin="${bins[0]}"
    if [[ ${#bins[@]} -gt 1 ]]; then
        echo ""
        header "Multiple executables found"
        local i=1
        for b in "${bins[@]}"; do
            echo -e "  ${BOLD}${i})${RESET} $(basename "$b")  ${DIM}${b}${RESET}"
            ((i++))
        done
        echo ""
        local choice
        read -rp "$(echo -e "${CYAN}${BOLD}Which to install? [1-${#bins[@]}]:${RESET} ")" choice
        if [[ "$choice" =~ ^[0-9]+$ ]] && (( choice >= 1 && choice <= ${#bins[@]} )); then
            chosen_bin="${bins[$((choice-1))]}"
        fi
    fi

    local install_dir="$REAL_HOME/.local/bin"
    mkdir -p "$install_dir"
    local bin_name
    bin_name=$(basename "$chosen_bin")
    cp "$chosen_bin" "$install_dir/$bin_name"
    chmod +x "$install_dir/$bin_name"
    rm -rf "$tmpdir"

    registry_add "$pkg_name" "archive" "$install_dir/$bin_name"
    success "Installed ${BOLD}${bin_name}${RESET} → ${install_dir}/${bin_name}"

    if [[ ":$PATH:" != *":$install_dir:"* ]]; then
        warn "~/.local/bin is not in PATH. Add to ~/.bashrc or ~/.zshrc:"
        echo -e "  ${DIM}export PATH=\"\$HOME/.local/bin:\$PATH\"${RESET}"
    fi
}

# ── Search functions ─────────────────────────────────────────
search_apt() {
    command -v apt-cache &>/dev/null && apt-cache show "$1" &>/dev/null && echo "apt"
}

search_flatpak() {
    has_flatpak && flatpak search --columns=application "$1" 2>/dev/null | grep -qi "$1" && echo "flatpak"
}

search_snap() {
    has_snap && snap find "$1" 2>/dev/null | grep -qi "$1" && echo "snap"
}

search_github() {
    local pkg="$1"
    local json
    json=$(curl -sf --max-time 8 \
        -H "Accept: application/vnd.github+json" \
        "https://api.github.com/search/repositories?q=${pkg}+in:name&sort=stars&per_page=5" 2>/dev/null) || return 1
    [[ -z "$json" ]] && return 1
    python3 - "$json" "$pkg" << 'PYEOF'
import json, sys
data = json.loads(sys.argv[1])
query = sys.argv[2].lower()
items = data.get('items', [])
if not items:
    sys.exit(1)
for item in items:
    if item['name'].lower() == query:
        print(item['full_name'])
        sys.exit(0)
print(items[0]['full_name'])
PYEOF
}

# ── AppImage helpers ─────────────────────────────────────────
install_appimage() {
    local src="$1"
    local name

    if [[ "$src" =~ ^https?:// ]]; then
        name=$(basename "$src" | sed 's/\.AppImage$//i; s/\.appimage$//i')
        info "Downloading AppImage..."
        curl -Lf# "$src" -o "$UNI_APPIMAGE_DIR/${name}.AppImage" || die "Failed to download AppImage"
        src="$UNI_APPIMAGE_DIR/${name}.AppImage"
    else
        [[ -f "$src" ]] || die "File not found: $src"
        name=$(basename "$src" | sed 's/\.AppImage$//i; s/\.appimage$//i')
        cp "$src" "$UNI_APPIMAGE_DIR/${name}.AppImage"
    fi

    chmod +x "$UNI_APPIMAGE_DIR/${name}.AppImage"

    local icon_path=""
    step "Extracting icon..."
    local tmpdir
    tmpdir=$(mktemp -d)
    pushd "$tmpdir" &>/dev/null || die "mktemp failed"
    if "$UNI_APPIMAGE_DIR/${name}.AppImage" --appimage-extract &>/dev/null; then
        local extracted_icon
        extracted_icon=$(find squashfs-root -name "*.png" -o -name "*.svg" 2>/dev/null \
            | xargs ls -S 2>/dev/null | head -1)
        if [[ -n "$extracted_icon" ]]; then
            local icon_ext="${extracted_icon##*.}"
            cp "$extracted_icon" "$UNI_ICON_DIR/${name}.${icon_ext}"
            icon_path="$UNI_ICON_DIR/${name}.${icon_ext}"
        fi
    fi
    popd &>/dev/null || true
    rm -rf "$tmpdir"

    [[ -z "$icon_path" ]] && icon_path="application-x-executable"

    cat > "$UNI_DESKTOP_DIR/${name}.desktop" <<EOF
[Desktop Entry]
Name=${name}
Exec=$UNI_APPIMAGE_DIR/${name}.AppImage
Icon=${icon_path}
Type=Application
Categories=Utility;
Comment=Installed via uni
EOF

    registry_add "$name" "appimage" "$UNI_APPIMAGE_DIR/${name}.AppImage"
    success "AppImage installed: ${BOLD}${name}${RESET}"
    success "Desktop entry: ${UNI_DESKTOP_DIR}/${name}.desktop"
}

remove_appimage() {
    local name="$1"
    local appimage_path
    appimage_path=$(registry_get "$name" "id")
    [[ -f "$appimage_path" ]] && rm -f "$appimage_path" && success "Removed AppImage"
    [[ -f "$UNI_DESKTOP_DIR/${name}.desktop" ]] && rm -f "$UNI_DESKTOP_DIR/${name}.desktop" && success "Removed desktop entry"
    for ext in png svg; do
        [[ -f "$UNI_ICON_DIR/${name}.${ext}" ]] && rm -f "$UNI_ICON_DIR/${name}.${ext}"
    done
    registry_remove "$name"
}

# ── Install deb ──────────────────────────────────────────────
install_deb() {
    local file="$1"
    [[ -f "$file" ]] || die "File not found: $file"

    # Architecture check — fail early with clear message
    local deb_arch sys
    deb_arch=$(dpkg-deb --field "$file" Architecture 2>/dev/null || true)
    sys=$(sys_arch)
    if [[ -n "$deb_arch" && "$deb_arch" != "all" && "$deb_arch" != "$sys" ]]; then
        spin_stop
        die "Architecture mismatch: package is ${BOLD}${deb_arch}${RESET} but system is ${BOLD}${sys}${RESET}.\nDownload the ${sys} build instead."
    fi

    local name
    name=$(dpkg-deb --field "$file" Package 2>/dev/null || basename "$file" .deb)
    info "Installing .deb: ${BOLD}${name}${RESET}"

    if sudo dpkg -i "$file"; then
        registry_add "$name" "dpkg" "$name"
        success "Installed ${BOLD}${name}${RESET} via dpkg"
    else
        warn "dpkg errors — fixing dependencies..."
        sudo "$(apt_cmd)" -f install -y
        if dpkg -s "$name" 2>/dev/null | grep -q "^Status: install ok installed"; then
            registry_add "$name" "dpkg" "$name"
            success "Installed ${BOLD}${name}${RESET} via dpkg (after fix)"
        else
            die "Installation of ${name} failed."
        fi
    fi
}

# ── Installed detection ──────────────────────────────────────
is_installed_apt() {
    dpkg -s "$1" &>/dev/null && dpkg -s "$1" 2>/dev/null | grep -q "^Status: install ok installed"
}

is_installed_flatpak() {
    has_flatpak && flatpak list --columns=application,name 2>/dev/null | grep -qi "$1"
}

is_installed_snap() {
    has_snap && snap list "$1" &>/dev/null
}

UNI_INSTALLED_IN=()
detect_installed_backends() {
    local pkg="$1"
    UNI_INSTALLED_IN=()
    { is_installed_apt     "$pkg" && UNI_INSTALLED_IN+=("apt");     } 2>/dev/null || true
    { is_installed_flatpak "$pkg" && UNI_INSTALLED_IN+=("flatpak"); } 2>/dev/null || true
    { is_installed_snap    "$pkg" && UNI_INSTALLED_IN+=("snap");    } 2>/dev/null || true
    local reg_backend
    reg_backend=$(registry_get "$pkg" "backend") || true
    if [[ -n "$reg_backend" ]] && ! in_array "$reg_backend" "${UNI_INSTALLED_IN[@]:-}"; then
        UNI_INSTALLED_IN+=("${reg_backend}(uni)")
    fi
}

# ── Already-installed prompt ─────────────────────────────────
UNI_DECISION=""
handle_already_installed() {
    local pkg="$1"; shift
    local installed_str
    installed_str=$(IFS=', '; echo "$*")

    echo ""
    echo -e "${YELLOW}${BOLD} ⚠  ${pkg}${RESET}${YELLOW} already installed${RESET} via ${BOLD}${installed_str}${RESET}"
    divider

    PS3="$(echo -e "${CYAN}${BOLD}What to do?${RESET} [1-3]: ")"
    select _ in "Reinstall / refresh" "Install from a different source" "Exit — keep as is"; do
        case "$REPLY" in
            1) UNI_DECISION="reinstall"; break ;;
            2) UNI_DECISION="alongside"; break ;;
            3) UNI_DECISION="exit";      break ;;
            *) warn "Enter 1, 2, or 3." ;;
        esac
    done
}

# ── Source selection menu ────────────────────────────────────
select_source() {
    local pkg="$1"; shift
    echo ""
    echo -e "${BOLD}${BLUE}${pkg}${RESET} found in multiple sources:"
    divider
    PS3="$(echo -e "${CYAN}${BOLD}Select source${RESET} [1-$#]: ")"
    local chosen
    select chosen in "$@"; do
        [[ -n "$chosen" ]] && { echo "$chosen"; return; }
        warn "Invalid choice."
    done
}

# ── Resolve flatpak App ID ───────────────────────────────────
resolve_flatpak_id() {
    local pkg="$1"
    [[ "$pkg" == *.*.* ]] && { echo "$pkg"; return 0; }

    local results
    results=$(flatpak search --columns=application,name "$pkg" 2>/dev/null) || true
    [[ -z "$results" ]] && return 1

    local filtered
    filtered=$(echo "$results" | grep -iv "plugin\|runtime\|extension" | grep -i "$pkg") || true
    [[ -z "$filtered" ]] && filtered=$(echo "$results" | grep -i "$pkg")
    [[ -z "$filtered" ]] && return 1

    local exact
    exact=$(echo "$filtered" | awk -F'\t' -v p="${pkg,,}" 'tolower($2)==p{print $1}' | head -1)
    [[ -n "$exact" ]] && { echo "$exact"; return 0; }

    local id_match
    id_match=$(echo "$filtered" | awk -F'\t' -v p="${pkg,,}" 'tolower($1)~("\\."p"$"){print $1}' | head -1)
    [[ -n "$id_match" ]] && { echo "$id_match"; return 0; }

    local ids=() names=()
    while IFS=$'\t' read -r appid appname _; do
        [[ -z "$appid" ]] && continue
        ids+=("$appid")
        names+=("${appname:-$appid}")
    done <<< "$filtered"
    [[ ${#ids[@]} -eq 1 ]] && { echo "${ids[0]}"; return 0; }

    echo "" >&2
    echo -e "${BOLD}${BLUE}Multiple flatpak matches for '${pkg}':${RESET}" >&2
    divider >&2
    PS3="$(echo -e "${CYAN}${BOLD}Select:${RESET} ")"
    local fchoice
    select fchoice in "${names[@]}"; do
        [[ -n "$fchoice" ]] && { echo "${ids[$((REPLY-1))]}"; return 0; }
        warn "Invalid choice." >&2
    done
}

# ── Install executor ─────────────────────────────────────────
do_install() {
    local pkg="$1" chosen_source="$2"
    case "$chosen_source" in
        apt)
            sudo "$(apt_cmd)" install -y "$pkg"
            registry_add "$pkg" "apt" "$pkg"
            success "Installed ${BOLD}${pkg}${RESET} via $(apt_cmd)"
            ;;
        flatpak)
            local appid
            appid=$(resolve_flatpak_id "$pkg")
            [[ -z "$appid" ]] && die "Could not resolve flatpak ID for '${pkg}'"
            flatpak install --noninteractive -y --or-update "$appid" \
                || die "Failed to install flatpak ${appid}"
            registry_add "$pkg" "flatpak" "$appid"
            success "Installed ${BOLD}${pkg}${RESET} via flatpak (${appid})"
            ;;
        snap)
            sudo snap install "$pkg" || die "Failed to install snap ${pkg}"
            registry_add "$pkg" "snap" "$pkg"
            success "Installed ${BOLD}${pkg}${RESET} via snap"
            ;;
    esac
}

# ── INSTALL ──────────────────────────────────────────────────
GH_REPO=""
cmd_install() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni install <package|user/repo|github-url|file.deb|file.appimage|url>"

    init_registry

    # GitHub URL or user/repo shorthand — check first before anything else
    local maybe_repo
    maybe_repo=$(parse_github_repo "$pkg")
    if [[ -n "$maybe_repo" ]]; then
        GH_REPO="$maybe_repo"
        install_from_github "$GH_REPO"
        return
    fi

    # Local or remote .deb
    if [[ "$pkg" == *.deb ]]; then
        if [[ "$pkg" =~ ^https?:// ]]; then
            local tmp_deb
            tmp_deb=$(mktemp --suffix=.deb)
            info "Downloading .deb..."
            curl -Lf# "$pkg" -o "$tmp_deb" || { rm -f "$tmp_deb"; die "Download failed."; }
            install_deb "$tmp_deb"
            rm -f "$tmp_deb"
        else
            install_deb "$pkg"
        fi
        return
    fi

    # AppImage
    if [[ "$pkg" == *.AppImage || "$pkg" == *.appimage ]] || \
       [[ "$pkg" =~ ^https?://.+\.(AppImage|appimage)$ ]]; then
        install_appimage "$pkg"; return
    fi

    # Check already installed
    spin_start "Checking existing installations..."
    detect_installed_backends "$pkg"
    spin_stop

    local exclude_sources=()

    if [[ ${#UNI_INSTALLED_IN[@]} -gt 0 ]]; then
        handle_already_installed "$pkg" "${UNI_INSTALLED_IN[@]}"
        case "$UNI_DECISION" in
            exit)
                info "Nothing changed."
                return 0
                ;;
            reinstall)
                local pb="${UNI_INSTALLED_IN[0]}"
                pb="${pb%(uni)}"
                info "Reinstalling ${BOLD}${pkg}${RESET} from ${pb}..."
                case "$pb" in
                    apt)
                        sudo "$(apt_cmd)" install --reinstall -y "$pkg"
                        registry_add "$pkg" "apt" "$pkg"
                        success "Reinstalled ${BOLD}${pkg}${RESET}"
                        ;;
                    flatpak)
                        local appid
                        appid=$(registry_get "$pkg" "id")
                        [[ -z "$appid" ]] && appid=$(resolve_flatpak_id "$pkg")
                        flatpak update --noninteractive -y "$appid" || \
                            flatpak install --noninteractive -y "$appid"
                        registry_add "$pkg" "flatpak" "$appid"
                        success "Reinstalled ${BOLD}${pkg}${RESET}"
                        ;;
                    snap)
                        sudo snap refresh "$pkg" || sudo snap install "$pkg"
                        success "Reinstalled ${BOLD}${pkg}${RESET}"
                        ;;
                    dpkg|archive|appimage)
                        warn "Re-run: uni install <url-or-file> to reinstall this package."
                        ;;
                esac
                return 0
                ;;
            alongside)
                for b in "${UNI_INSTALLED_IN[@]}"; do
                    exclude_sources+=("${b%(uni)}")
                done
                info "Searching other sources (excluding: ${exclude_sources[*]})..."
                ;;
        esac
    fi

    # Search backends sequentially with spinners
    local sources=()
    local apt_result="" flatpak_result="" snap_result="" gh_result=""

    spin_start "Searching apt..."
    apt_result=$(search_apt "$pkg" 2>/dev/null) || true
    spin_stop

    spin_start "Searching flatpak..."
    flatpak_result=$(search_flatpak "$pkg" 2>/dev/null) || true
    spin_stop

    spin_start "Searching snap..."
    snap_result=$(search_snap "$pkg" 2>/dev/null) || true
    spin_stop

    spin_start "Searching GitHub..."
    gh_result=$(search_github "$pkg" 2>/dev/null) || true
    spin_stop

    [[ -n "$apt_result"     ]] && ! in_array "apt"     "${exclude_sources[@]:-}" && sources+=("apt")
    [[ -n "$flatpak_result" ]] && ! in_array "flatpak" "${exclude_sources[@]:-}" && sources+=("flatpak")
    [[ -n "$snap_result"    ]] && ! in_array "snap"    "${exclude_sources[@]:-}" && sources+=("snap")
    [[ -n "$gh_result"      ]] && sources+=("github (${gh_result})")

    if [[ ${#sources[@]} -eq 0 ]]; then
        if [[ ${#exclude_sources[@]} -gt 0 ]]; then
            warn "No other sources found for '${pkg}'."
        else
            warn "Package '${pkg}' not found in apt, flatpak, snap, or GitHub."
            echo -e "  ${DIM}Try: uni install user/repo  |  uni install <url.deb>  |  uni install <url.AppImage>${RESET}"
        fi
        exit 1
    fi

    local chosen_source
    if [[ ${#sources[@]} -eq 1 ]]; then
        chosen_source="${sources[0]}"
        info "Found in ${BOLD}${chosen_source}${RESET}"
    else
        chosen_source=$(select_source "$pkg" "${sources[@]}")
    fi

    if [[ "$chosen_source" == github* ]]; then
        GH_REPO="$gh_result"
        install_from_github "$GH_REPO"
    else
        do_install "$pkg" "$chosen_source"
    fi
}

# ── REMOVE ───────────────────────────────────────────────────
cmd_remove() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni remove <package>"
    init_registry

    local backend id
    backend=$(registry_get "$pkg" "backend")
    id=$(registry_get "$pkg" "id")

    if [[ -z "$backend" ]]; then
        warn "'${pkg}' not in uni registry — trying all backends..."
        if is_installed_apt "$pkg"; then
            sudo "$(apt_cmd)" remove -y "$pkg"
        elif is_installed_flatpak "$pkg"; then
            local appid
            appid=$(flatpak list --columns=application,name 2>/dev/null | grep -i "$pkg" | head -1 | awk '{print $1}')
            [[ -n "$appid" ]] \
                && flatpak uninstall --noninteractive -y "$appid" \
                || flatpak uninstall -y "$pkg"
        elif is_installed_snap "$pkg"; then
            sudo snap remove "$pkg"
        else
            die "'${pkg}' not found in any backend."
        fi
        return
    fi

    info "Removing ${BOLD}${pkg}${RESET} [${backend}]..."

    case "$backend" in
        apt)      sudo "$(apt_cmd)" remove -y "$id" ;;
        dpkg)     sudo dpkg -r "$id" || sudo "$(apt_cmd)" remove -y "$id" ;;
        flatpak)  flatpak uninstall --noninteractive -y "$id" ;;
        snap)     sudo snap remove "$id" ;;
        appimage) remove_appimage "$pkg"; return ;;
        archive)
            [[ -f "$id" ]] && rm -f "$id" && success "Removed binary: ${id}"
            registry_remove "$pkg"; return
            ;;
    esac

    registry_remove "$pkg"
    success "Removed ${BOLD}${pkg}${RESET}"
}

# ── PURGE ────────────────────────────────────────────────────
cmd_purge() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni purge <package>"
    init_registry

    local backend id
    backend=$(registry_get "$pkg" "backend")
    id=$(registry_get "$pkg" "id")

    if [[ -z "$backend" ]]; then
        warn "'${pkg}' not in uni registry — trying all backends..."
        if is_installed_apt "$pkg"; then
            sudo "$(apt_cmd)" purge -y "$pkg" && sudo "$(apt_cmd)" autoremove -y
        elif is_installed_flatpak "$pkg"; then
            local appid
            appid=$(flatpak list --columns=application,name 2>/dev/null | grep -i "$pkg" | head -1 | awk '{print $1}')
            if [[ -n "$appid" ]]; then
                flatpak uninstall --noninteractive -y "$appid"
                rm -rf "$REAL_HOME/.var/app/$appid"
            fi
        elif is_installed_snap "$pkg"; then
            sudo snap remove --purge "$pkg"
        else
            die "'${pkg}' not found in any backend."
        fi
        return
    fi

    info "Purging ${BOLD}${pkg}${RESET} [${backend}] + config/data..."

    case "$backend" in
        apt|dpkg)
            sudo "$(apt_cmd)" purge -y "$id"
            sudo "$(apt_cmd)" autoremove -y
            ;;
        flatpak)
            flatpak uninstall --noninteractive -y "$id"
            local flatpak_data="$REAL_HOME/.var/app/$id"
            [[ -d "$flatpak_data" ]] && rm -rf "$flatpak_data" && success "Removed app data: ${flatpak_data}"
            ;;
        snap)     sudo snap remove --purge "$id" ;;
        appimage) remove_appimage "$pkg"; return ;;
        archive)
            [[ -f "$id" ]] && rm -f "$id" && success "Removed binary: ${id}"
            registry_remove "$pkg"; return
            ;;
    esac

    registry_remove "$pkg"
    success "Purged ${BOLD}${pkg}${RESET}"
}

# ── LIST ─────────────────────────────────────────────────────
cmd_list() {
    init_registry
    header "Packages installed via uni"
    registry_list
    divider
    echo ""
}

# ── UPDATE ───────────────────────────────────────────────────
cmd_update() {
    init_registry
    header "Updating all backends"

    if has_apt; then
        info "apt / nala..."
        sudo "$(apt_cmd)" update && sudo "$(apt_cmd)" upgrade -y
        echo ""
    fi

    if has_flatpak; then
        info "Flatpak..."
        flatpak update --noninteractive -y
        echo ""
    fi

    if has_snap; then
        info "Snap..."
        sudo snap refresh
        echo ""
    fi

    success "All backends updated."
}

# ── HELP ─────────────────────────────────────────────────────
cmd_help() {
    echo ""
    echo -e "${BOLD}uni${RESET} ${DIM}v${UNI_VERSION}${RESET} — Universal Package Manager"

    echo -e "  ${BOLD}uni install${RESET} <url.deb>            Download & install .deb"
    echo -e "  ${BOLD}uni install${RESET} <file.appimage>      Install AppImage + desktop entry"
    echo -e "  ${BOLD}uni install${RESET} <url.AppImage>       Download & install AppImage"

    echo -e "  ${BOLD}uni purge${RESET}   <package>            Remove + all config/data"

# ── Entry point ──────────────────────────────────────────────
main() {
    local cmd="${1:-help}"
    shift || true
    case "$cmd" in
        install) cmd_install "$@" ;;
        remove)  cmd_remove  "$@" ;;
        purge)   cmd_purge   "$@" ;;
        list)    cmd_list ;;
        update)  cmd_update ;;
        version) echo "uni v${UNI_VERSION}" ;;
        help|--help|-h) cmd_help ;;
        *) error "Unknown command: ${cmd}"; cmd_help; exit 1 ;;
    esac
}

main "$@"
