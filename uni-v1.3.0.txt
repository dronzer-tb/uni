#!/usr/bin/env bash
# ============================================================
#  uni — Universal Package Manager Wrapper
#  by Dronzer Studios / Kasniya
#  Commands: uni install | uni remove | uni purge | uni list | uni update
# ============================================================

# Interactive script — no pipefail/errexit, explicit handling only
set -u

UNI_VERSION="1.3.0"

# Always resolve to the real invoking user's home, even under sudo
if [[ -n "${SUDO_USER:-}" ]]; then
    REAL_HOME=$(getent passwd "$SUDO_USER" | cut -d: -f6)
else
    REAL_HOME="$HOME"
fi

UNI_REGISTRY="$REAL_HOME/.local/share/uni/installed.json"
UNI_APPIMAGE_DIR="$REAL_HOME/.local/share/uni/appimages"
UNI_DESKTOP_DIR="$REAL_HOME/.local/share/applications"
UNI_ICON_DIR="$REAL_HOME/.local/share/uni/icons"

# ── Colors ──────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
DIM='\033[2m'
RESET='\033[0m'

# ── Helpers ──────────────────────────────────────────────────
info()    { echo -e "${CYAN}${BOLD}::${RESET} $*"; }
success() { echo -e "${GREEN}${BOLD}✓${RESET} $*"; }
warn()    { echo -e "${YELLOW}${BOLD}!${RESET} $*"; }
error()   { echo -e "${RED}${BOLD}✗${RESET} $*" >&2; }
die()     { error "$*"; exit 1; }

divider() { echo -e "${DIM}────────────────────────────────────────${RESET}"; }

in_array() {
    local needle="$1"; shift
    local e
    for e in "$@"; do [[ "$e" == "$needle" ]] && return 0; done
    return 1
}

# ── Dependency Checks ────────────────────────────────────────
command -v python3 &>/dev/null || die "python3 is required but not installed. Please install it to continue."

# Warn if running via sudo
if [[ -n "${SUDO_USER:-}" ]]; then
    warn "Running uni with sudo is generally not recommended."
    warn "User-level backends (like Flatpak) may not be found because sudo uses root's environment."
    warn "Uni will prompt for sudo internally when needed for package operations."
fi

# ── Init registry ────────────────────────────────────────────
init_registry() {
    mkdir -p "$(dirname "$UNI_REGISTRY")" "$UNI_APPIMAGE_DIR" "$UNI_DESKTOP_DIR" "$UNI_ICON_DIR"
    [[ -f "$UNI_REGISTRY" ]] || echo '{}' > "$UNI_REGISTRY"
    # Fix ownership if running as sudo
    if [[ -n "${SUDO_USER:-}" ]]; then
        chown -R "$SUDO_USER:$SUDO_USER" "$(dirname "$UNI_REGISTRY")" 2>/dev/null || true
    fi
}

# ── Registry helpers (pure bash + python3) ───────────────────
registry_add() {
    local name="$1" backend="$2" id="$3"
    python3 - "$name" "$backend" "$id" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, backend, id_val, reg_path = sys.argv[1], sys.argv[2], sys.argv[3], sys.argv[4]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
d[name] = {'backend': backend, 'id': id_val}
with open(reg_path, 'w') as f: json.dump(d, f, indent=2)
PYEOF
}

registry_remove() {
    local name="$1"
    python3 - "$name" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, reg_path = sys.argv[1], sys.argv[2]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
d.pop(name, None)
with open(reg_path, 'w') as f: json.dump(d, f, indent=2)
PYEOF
}

registry_get() {
    local name="$1" field="$2"
    python3 - "$name" "$field" "$UNI_REGISTRY" << 'PYEOF'
import json, sys
name, field, reg_path = sys.argv[1], sys.argv[2], sys.argv[3]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
print(d.get(name, {}).get(field, ''))
PYEOF
}

registry_list() {
    python3 - "$UNI_REGISTRY" << 'PYEOF'
import json, sys
reg_path = sys.argv[1]
try:
    with open(reg_path) as f: d = json.load(f)
except (FileNotFoundError, json.JSONDecodeError):
    d = {}
if not d:
    print('  No packages installed via uni.')
else:
    for name, info in sorted(d.items()):
        print(f"  {name:<30} [{info['backend']}]  {info['id']}")
PYEOF
}

# ── Backend detection ────────────────────────────────────────
has_nala()    { command -v nala    &>/dev/null; }
has_apt()     { command -v apt     &>/dev/null; }
has_flatpak() { command -v flatpak &>/dev/null; }
has_snap()    { command -v snap    &>/dev/null; }
has_dpkg()    { command -v dpkg    &>/dev/null; }

apt_cmd() { has_nala && echo "nala" || echo "apt"; }

# ── Search functions ─────────────────────────────────────────
search_apt() {
    local pkg="$1"
    if command -v apt-cache &>/dev/null; then
        if apt-cache show "$pkg" &>/dev/null; then echo "apt"; fi
    fi
}

search_flatpak() {
    local pkg="$1"
    if has_flatpak; then
        if flatpak search --columns=application "$pkg" 2>/dev/null | grep -qi "$pkg"; then
            echo "flatpak"
        fi
    fi
}

search_snap() {
    local pkg="$1"
    if has_snap; then
        if snap find "$pkg" 2>/dev/null | grep -qi "$pkg"; then
            echo "snap"
        fi
    fi
}

# ── AppImage helpers ─────────────────────────────────────────
install_appimage() {
    local src="$1"
    local name

    # If it's a URL, download it
    if [[ "$src" =~ ^https?:// ]]; then
        name=$(basename "$src" | sed 's/\.AppImage$//i; s/\.appimage$//i')
        info "Downloading AppImage from $src..."
        curl -Lf "$src" -o "$UNI_APPIMAGE_DIR/${name}.AppImage" || die "Failed to download AppImage"
        src="$UNI_APPIMAGE_DIR/${name}.AppImage"
    else
        [[ -f "$src" ]] || die "File not found: $src"
        name=$(basename "$src" | sed 's/\.AppImage$//i; s/\.appimage$//i')
        cp "$src" "$UNI_APPIMAGE_DIR/${name}.AppImage"
    fi

    chmod +x "$UNI_APPIMAGE_DIR/${name}.AppImage"

    # Try to extract icon
    local icon_path=""
    info "Attempting to extract icon from AppImage..."
    
    local tmpdir
    tmpdir=$(mktemp -d)
    pushd "$tmpdir" &>/dev/null || die "Failed to create temp dir"
    if "$UNI_APPIMAGE_DIR/${name}.AppImage" --appimage-extract &>/dev/null; then
        local extracted_icon
        extracted_icon=$(find squashfs-root -name "*.png" -o -name "*.svg" 2>/dev/null | xargs ls -S 2>/dev/null | head -1)
        if [[ -n "$extracted_icon" ]]; then
            local icon_ext="${extracted_icon##*.}"
            local icon_dest="$UNI_ICON_DIR/${name}.${icon_ext}"
            cp "$extracted_icon" "$icon_dest"
            icon_path="$icon_dest"
        fi
    fi
    popd &>/dev/null || true
    rm -rf "$tmpdir"

    [[ -z "$icon_path" ]] && icon_path="application-x-executable"

    # Create .desktop entry
    cat > "$UNI_DESKTOP_DIR/${name}.desktop" <<EOF
[Desktop Entry]
Name=${name}
Exec=$UNI_APPIMAGE_DIR/${name}.AppImage
Icon=${icon_path}
Type=Application
Categories=Utility;
Comment=Installed via uni
EOF

    success "AppImage installed: ${name}"
    success "Desktop entry created at $UNI_DESKTOP_DIR/${name}.desktop"
    registry_add "$name" "appimage" "$UNI_APPIMAGE_DIR/${name}.AppImage"
}

remove_appimage() {
    local name="$1"
    local appimage_path
    appimage_path=$(registry_get "$name" "id")
    
    [[ -f "$appimage_path" ]] && rm -f "$appimage_path" && success "Removed AppImage: $appimage_path"
    [[ -f "$UNI_DESKTOP_DIR/${name}.desktop" ]] && rm -f "$UNI_DESKTOP_DIR/${name}.desktop" && success "Removed desktop entry"
    
    for ext in png svg; do
        local icon_file="$UNI_ICON_DIR/${name}.${ext}"
        if [[ -f "$icon_file" ]]; then
            rm -f "$icon_file"
            success "Removed icon: $icon_file"
        fi
    done

    registry_remove "$name"
}

# ── Install deb file ─────────────────────────────────────────
install_deb() {
    local file="$1"
    [[ -f "$file" ]] || die "File not found: $file"
    local name
    name=$(dpkg-deb --field "$file" Package 2>/dev/null || basename "$file" .deb)
    info "Installing .deb package: $name"
    sudo dpkg -i "$file" || sudo "$(apt_cmd)" -f install -y
    registry_add "$name" "dpkg" "$name"
    success "Installed $name via dpkg"
}

# ── Already-installed detection ─────────────────────────────
is_installed_apt() {
    dpkg -s "$1" &>/dev/null && dpkg -s "$1" 2>/dev/null | grep -q "^Status: install ok installed" || return 1
}

is_installed_flatpak() {
    local pkg="$1"
    if has_flatpak; then
        # Check both App ID and Name columns case-insensitively
        flatpak list --columns=application,name 2>/dev/null | grep -qi "$pkg" || return 1
    else
        return 1
    fi
}

is_installed_snap() {
    if has_snap; then
        snap list "$1" &>/dev/null || return 1
    else
        return 1
    fi
}

# Populates global UNI_INSTALLED_IN array with backends where pkg is found
UNI_INSTALLED_IN=()
detect_installed_backends() {
    local pkg="$1"
    UNI_INSTALLED_IN=()
    { is_installed_apt "$pkg"      && UNI_INSTALLED_IN+=("apt");     } 2>/dev/null || true
    { is_installed_flatpak "$pkg"  && UNI_INSTALLED_IN+=("flatpak"); } 2>/dev/null || true
    { is_installed_snap "$pkg"     && UNI_INSTALLED_IN+=("snap");    } 2>/dev/null || true
    
    local reg_backend
    reg_backend=$(registry_get "$pkg" "backend") || true
    if [[ -n "$reg_backend" ]] && [[ ! " ${UNI_INSTALLED_IN[*]:-} " =~ " $reg_backend " ]]; then
        UNI_INSTALLED_IN+=("${reg_backend}(uni-registry)")
    fi
}

# ── Already-installed prompt ─────────────────────────────────
UNI_DECISION=""
handle_already_installed() {
    local pkg="$1"
    shift
    local installed_in=("$@")
    local installed_str
    installed_str=$(IFS=','; echo "${installed_in[*]}")

    echo ""
    echo -e "${YELLOW}${BOLD}⚠ ${pkg}${RESET}${YELLOW} is already installed${RESET} via: ${BOLD}${installed_str}${RESET}"
    divider

    local choice
    PS3="What would you like to do? [1-3]: "
    select choice in "Reinstall / refresh from same source" "Install alongside from a different source" "Exit — keep what I have"; do
        case "$REPLY" in
            1) UNI_DECISION="reinstall"; break ;;
            2) UNI_DECISION="alongside"; break ;;
            3) UNI_DECISION="exit"; break ;;
            *) warn "Invalid choice. Please enter 1, 2, or 3." ;;
        esac
    done
}

# ── Present source selection menu ────────────────────────────
select_source() {
    local pkg="$1"
    shift
    local sources=("$@")

    echo ""
    echo -e "${BOLD}${BLUE}${pkg}${RESET} is available at multiple sources:"
    divider

    local chosen_source
    PS3="Select installation source [1-${#sources[@]}] (Ctrl+C to cancel): "
    select chosen_source in "${sources[@]}"; do
        if [[ -n "$chosen_source" ]]; then
            echo "$chosen_source"
            return
        else
            warn "Invalid choice. Please enter a number between 1 and ${#sources[@]}."
        fi
    done
}

# ── Resolve best flatpak App ID ──────────────────────────────
resolve_flatpak_id() {
    local pkg="$1"

    # If it looks like a full reverse-domain ID already, trust it
    if [[ "$pkg" == *.*.* ]]; then
        echo "$pkg"; return 0
    fi

    # Search and get application IDs + names
    local results
    results=$(flatpak search --columns=application,name "$pkg" 2>/dev/null) || true
    [[ -z "$results" ]] && return 1

    # Filter: only lines where app ID does NOT contain Plugin/plugin/Runtime/runtime
    local filtered
    filtered=$(echo "$results" | grep -iv "plugin\|runtime\|extension" | grep -i "$pkg") || true
    [[ -z "$filtered" ]] && filtered=$(echo "$results" | grep -i "$pkg")
    [[ -z "$filtered" ]] && return 1

    # Try exact name match first (case-insensitive): pick app whose name == pkg
    local exact
    exact=$(echo "$filtered" | awk -F'\t' -v p="${pkg,,}" 'tolower($2) == p {print $1}' | head -1)
    if [[ -n "$exact" ]]; then
        echo "$exact"; return 0
    fi

    # Try app ID ending in .Name (e.g. org.videolan.VLC for "vlc")
    local id_match
    id_match=$(echo "$filtered" | awk -F'\t' -v p="${pkg,,}" 'tolower($1) ~ ("\\." p "$") {print $1}' | head -1)
    if [[ -n "$id_match" ]]; then
        echo "$id_match"; return 0
    fi

    # Multiple results — show user a menu
    local ids=()
    local names=()
    while IFS=$'\t' read -r appid appname _; do
        [[ -z "$appid" ]] && continue
        ids+=("$appid")
        names+=("${appname:-$appid}") # Fallback to appid if name is empty
    done <<< "$filtered"

    if [[ ${#ids[@]} -eq 1 ]]; then
        echo "${ids[0]}"; return 0
    fi

    echo "" >&2
    echo -e "${BOLD}${BLUE}Multiple flatpak matches found for '${pkg}':${RESET}" >&2
    divider >&2

    local fchoice
    PS3="Select the correct package [1-${#ids[@]}]: "
    select fchoice in "${names[@]}"; do
        if [[ -n "$fchoice" ]]; then
            echo "${ids[$((REPLY-1))]}"
            return 0
        else
            warn "Invalid choice. Please try again."
        fi
    done
}

# ── Shared install executor ───────────────────────────────────
do_install() {
    local pkg="$1" chosen_source="$2"
    case "$chosen_source" in
        apt)
            sudo "$(apt_cmd)" install -y "$pkg"
            registry_add "$pkg" "apt" "$pkg"
            success "Installed ${pkg} via $(apt_cmd)"
            ;;
        flatpak)
            local appid
            appid=$(resolve_flatpak_id "$pkg")
            [[ -z "$appid" ]] && die "Could not resolve flatpak app ID for '${pkg}'"
            # Removed explicit 'flathub' so it searches all system remotes
            flatpak install -y "$appid" || die "Failed to install flatpak $appid"
            registry_add "$pkg" "flatpak" "$appid"
            success "Installed ${pkg} via flatpak ($appid)"
            ;;
        snap)
            sudo snap install "$pkg" || die "Failed to install snap $pkg"
            registry_add "$pkg" "snap" "$pkg"
            success "Installed ${pkg} via snap"
            ;;
    esac
}

# ── INSTALL ──────────────────────────────────────────────────
cmd_install() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni install <package|file.deb|file.appimage|url>"

    init_registry

    # .deb file or URL pointing to .deb
    if [[ "$pkg" == *.deb ]]; then
        if [[ "$pkg" =~ ^https?:// ]]; then
            local tmp_deb
            tmp_deb=$(mktemp --suffix=.deb)
            info "Downloading .deb file from $pkg..."
            curl -Lf "$pkg" -o "$tmp_deb" || { rm -f "$tmp_deb"; die "Failed to download .deb file"; }
            install_deb "$tmp_deb"
            rm -f "$tmp_deb"
        else
            install_deb "$pkg"
        fi
        return
    fi

    # AppImage file or URL
    if [[ "$pkg" == *.AppImage ]] || [[ "$pkg" == *.appimage ]] || [[ "$pkg" =~ ^https?://.+\.(AppImage|appimage)$ ]]; then
        install_appimage "$pkg"; return
    fi

    # Check if already installed anywhere
    info "Searching for ${BOLD}${pkg}${RESET} across all backends..."
    detect_installed_backends "$pkg"

    local exclude_sources=()

    if [[ ${#UNI_INSTALLED_IN[@]} -gt 0 ]]; then
        handle_already_installed "$pkg" "${UNI_INSTALLED_IN[@]}"

        case "$UNI_DECISION" in
            exit)
                info "Keeping existing installation. Nothing changed."
                return 0
                ;;
            reinstall)
                local primary_backend="${UNI_INSTALLED_IN[0]}"
                primary_backend="${primary_backend%(uni-registry)}"
                info "Reinstalling ${BOLD}${pkg}${RESET} from ${primary_backend}..."
                case "$primary_backend" in
                    apt)
                        sudo "$(apt_cmd)" install --reinstall -y "$pkg"
                        registry_add "$pkg" "apt" "$pkg"
                        success "Reinstalled ${pkg} via $(apt_cmd)"
                        ;;
                    flatpak)
                        local appid
                        appid=$(registry_get "$pkg" "id") || true
                        if [[ -z "$appid" ]]; then
                            appid=$(resolve_flatpak_id "$pkg")
                        fi
                        flatpak update -y "$appid" || flatpak install -y "$appid"
                        registry_add "$pkg" "flatpak" "$appid"
                        success "Reinstalled ${pkg} via flatpak"
                        ;;
                    snap)
                        sudo snap refresh "$pkg" || sudo snap install "$pkg"
                        success "Reinstalled ${pkg} via snap"
                        ;;
                esac
                return 0
                ;;
            alongside)
                for b in "${UNI_INSTALLED_IN[@]}"; do
                    exclude_sources+=("${b%(uni-registry)}")
                done
                echo ""
                info "Will search for other sources (excluding: ${exclude_sources[*]})..."
                ;;
        esac
    fi

    # Search available backends
    local sources=()
    local apt_result="" flatpak_result="" snap_result=""
    apt_result=$(search_apt "$pkg") || true
    flatpak_result=$(search_flatpak "$pkg") || true
    snap_result=$(search_snap "$pkg") || true

    if [[ -n "$apt_result" ]]     && ! in_array "apt"     "${exclude_sources[@]:-}"; then sources+=("apt");     fi
    if [[ -n "$flatpak_result" ]] && ! in_array "flatpak" "${exclude_sources[@]:-}"; then sources+=("flatpak"); fi
    if [[ -n "$snap_result" ]]    && ! in_array "snap"    "${exclude_sources[@]:-}"; then sources+=("snap");    fi

    if [[ ${#sources[@]} -eq 0 ]]; then
        if [[ ${#exclude_sources[@]} -gt 0 ]]; then
            warn "No other sources found for '${pkg}' besides what's already installed."
        else
            warn "Package '${pkg}' not found in apt, flatpak, or snap."
            echo -e "  ${DIM}You can also install via: uni install <file.appimage> | uni install <file.deb> | uni install <url>${RESET}"
        fi
        exit 1
    fi

    local chosen_source
    if [[ ${#sources[@]} -eq 1 ]]; then
        chosen_source="${sources[0]}"
        info "Found in ${BOLD}${chosen_source}${RESET}. Installing..."
    else
        chosen_source=$(select_source "$pkg" "${sources[@]}")
    fi

    do_install "$pkg" "$chosen_source"
}

# ── REMOVE ───────────────────────────────────────────────────
cmd_remove() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni remove <package>"

    init_registry

    local backend
    backend=$(registry_get "$pkg" "backend")
    local id
    id=$(registry_get "$pkg" "id")

    if [[ -z "$backend" ]]; then
        warn "'${pkg}' not found in uni registry. Attempting to remove via available backends..."
        if is_installed_apt "$pkg"; then
            sudo "$(apt_cmd)" remove -y "$pkg"
        elif is_installed_flatpak "$pkg"; then
            # Must resolve the App ID for Flatpak uninstall
            local appid
            appid=$(resolve_flatpak_id "$pkg" 2>/dev/null || flatpak list --columns=application,name 2>/dev/null | grep -i "$pkg" | head -1 | awk '{print $1}')
            if [[ -n "$appid" ]]; then
                flatpak uninstall -y "$appid"
            else
                flatpak uninstall -y "$pkg" # Fallback
            fi
        elif is_installed_snap "$pkg"; then
            sudo snap remove "$pkg"
        else
            die "Package '${pkg}' not found in any backend."
        fi
        return
    fi

    info "Removing ${BOLD}${pkg}${RESET} [installed via ${backend}]..."

    case "$backend" in
        apt)     sudo "$(apt_cmd)" remove -y "$id" ;;
        flatpak) flatpak uninstall -y "$id" ;;
        snap)    sudo snap remove "$id" ;;
        dpkg)    sudo "$(apt_cmd)" remove -y "$id" ;;
        appimage) remove_appimage "$pkg"; return ;;
    esac

    registry_remove "$pkg"
    success "Removed ${pkg}"
}

# ── PURGE ────────────────────────────────────────────────────
cmd_purge() {
    local pkg="${1:-}"
    [[ -z "$pkg" ]] && die "Usage: uni purge <package>"

    init_registry

    local backend
    backend=$(registry_get "$pkg" "backend")
    local id
    id=$(registry_get "$pkg" "id")

    if [[ -z "$backend" ]]; then
        warn "'${pkg}' not found in uni registry. Attempting to purge via available backends..."
        if is_installed_apt "$pkg"; then
            sudo "$(apt_cmd)" purge -y "$pkg" && sudo "$(apt_cmd)" autoremove -y
        elif is_installed_flatpak "$pkg"; then
            local appid
            appid=$(resolve_flatpak_id "$pkg" 2>/dev/null || flatpak list --columns=application,name 2>/dev/null | grep -i "$pkg" | head -1 | awk '{print $1}')
            if [[ -n "$appid" ]]; then
                flatpak uninstall -y "$appid"
                rm -rf "$REAL_HOME/.var/app/$appid"
            else
                flatpak uninstall -y "$pkg"
                rm -rf "$REAL_HOME/.var/app/$pkg"
            fi
        elif is_installed_snap "$pkg"; then
            sudo snap remove --purge "$pkg"
        else
            die "Package '${pkg}' not found in any backend."
        fi
        return
    fi

    info "Purging ${BOLD}${pkg}${RESET} [installed via ${backend}] including config/data..."

    case "$backend" in
        apt|dpkg)
            sudo "$(apt_cmd)" purge -y "$id"
            sudo "$(apt_cmd)" autoremove -y
            ;;
        flatpak)
            flatpak uninstall -y "$id"
            # Remove flatpak app data
            local flatpak_data="$REAL_HOME/.var/app/$id"
            if [[ -d "$flatpak_data" ]]; then
                rm -rf "$flatpak_data"
                success "Removed flatpak app data: $flatpak_data"
            fi
            ;;
        snap)
            sudo snap remove --purge "$id"
            ;;
        appimage)
            remove_appimage "$pkg"
            return
            ;;
    esac

    registry_remove "$pkg"
    success "Purged ${pkg} (packages + config + data)"
}

# ── LIST ─────────────────────────────────────────────────────
cmd_list() {
    init_registry
    echo ""
    echo -e "${BOLD}Packages installed via uni:${RESET}"
    divider
    registry_list
    divider
    echo ""
}

# ── UPDATE ───────────────────────────────────────────────────
cmd_update() {
    init_registry
    info "Updating all packages via available backends..."
    echo ""

    if has_apt; then
        info "Updating apt packages..."
        sudo "$(apt_cmd)" update && sudo "$(apt_cmd)" upgrade -y
    fi

    if has_flatpak; then
        info "Updating flatpak packages..."
        flatpak update -y
    fi

    if has_snap; then
        info "Updating snap packages..."
        sudo snap refresh
    fi

    success "All backends updated."
}

# ── HELP ─────────────────────────────────────────────────────
cmd_help() {
    echo ""
    echo -e "${BOLD}uni${RESET} ${DIM}v${UNI_VERSION}${RESET} — Universal Package Manager"
    echo -e "${DIM}by Dronzer Studios${RESET}"
    divider
    echo -e "  ${BOLD}uni install${RESET} <package>          Search all backends, pick source"
    echo -e "  ${BOLD}uni install${RESET} <file.deb>         Install a local .deb file"
    echo -e "  ${BOLD}uni install${RESET} <url.deb>          Download & install a .deb file"
    echo -e "  ${BOLD}uni install${RESET} <file.appimage>    Install AppImage + create desktop entry"
    echo -e "  ${BOLD}uni install${RESET} <url.AppImage>     Download & install AppImage from URL"
    echo -e "  ${BOLD}uni remove${RESET}  <package>          Remove package (keeps config)"
    echo -e "  ${BOLD}uni purge${RESET}   <package>          Remove package + all config/data"
    echo -e "  ${BOLD}uni list${RESET}                       List all uni-managed packages"
    echo -e "  ${BOLD}uni update${RESET}                     Update all backends"
    divider
    echo -e "  ${DIM}Backends: $(apt_cmd) · flatpak · snap · dpkg · appimage${RESET}"
    echo ""
}

# ── Entry point ──────────────────────────────────────────────
main() {
    local cmd="${1:-help}"
    shift || true

    case "$cmd" in
        install) cmd_install "$@" ;;
        remove)  cmd_remove  "$@" ;;
        purge)   cmd_purge   "$@" ;;
        list)    cmd_list ;;
        update)  cmd_update ;;
        version) echo "uni v${UNI_VERSION}" ;;
        help|--help|-h) cmd_help ;;
        *) error "Unknown command: $cmd"; cmd_help; exit 1 ;;
    esac
}

main "$@"
